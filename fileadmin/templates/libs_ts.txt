lib.termine = COA_INT
lib.termine {
  10 = CONTENT
  10 {
    table = tt_content
    select {
      pidInList = 31
      orderBy = terminzeit asc, sorting
      andWhere = terminzeit > UNIX_TIMESTAMP(NOW())
    }
    stdWrap.required = 1
    stdWrap.wrap = <div id="p0"><h2 class="text-center">geplant</h2></div><ul class="termin-liste">|</ul>
  }
  20 = CONTENT
  20 {
    table = tt_content
    select {
      pidInList = 31
      orderBy = terminzeit desc, sorting
      groupBy = FROM_UNIXTIME(terminzeit, '%Y')
      andWhere = NOT terminzeit > UNIX_TIMESTAMP(NOW())
    }
    renderObj = COA
    renderObj {
      5 = TEXT
      5.field = terminzeit
      5.strftime = %Y
      5.wrap = <div id="p|">
      6 < .5
      6.wrap = <h2 class="text-center">|</h2></div>
      10 = COA
      10 {
        10 < tt_content
        20 = CONTENT
        20 {
          table = tt_content
          select {
            pidInList = 31
            where = FROM_UNIXTIME(terminzeit, '%Y')=FROM_UNIXTIME({field:terminzeit}, '%Y')
            where.insertData = 1
            orderBy = terminzeit desc, sorting
            andWhere = NOT terminzeit > UNIX_TIMESTAMP(NOW())
          }
        }
        wrap = <ul class="termin-liste">|</ul>
      }
    }
  }
}
